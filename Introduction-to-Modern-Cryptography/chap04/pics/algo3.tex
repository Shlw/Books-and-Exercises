        \begin{algorithm}[H]
            \caption{Experiment $5$}
            \DontPrintSemicolon
            \SetKwFunction{enc}{Encrypt}
            \SetKwFunction{dec}{Decrypt}
            \SetKwProg{func}{Function}{}{}

            $C,T\gets\varnothing$\;
            \func{\enc{$m\in\bin^{n/2}$}}{
                $r\sim\bin^{n/2},u\gets m\|r$\;
                $c\sim\bin^n$\;
                $T[c]\gets u,C\gets C\cup\{c\}$\;
                \KwRet $c$\;
            }
            \func{\dec{$c\in\bin^{n}$}}{
                \If{$c\notin C$}{
                    $u\sim\bin^{n}$\;
                    $T[c]\gets u,C\gets C\cup\{c\}$\;
                }
                \KwRet first half of $T[c]$\;
            }
        \end{algorithm}
