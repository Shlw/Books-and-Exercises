\begin{exercise}{1}
    Define 
    $$
    f(\bm x)=\min_{\bm y\in U}\text{Dist}(\bm x,\bm y).
    $$
    Let $x_1,\cdots,x_n$ be iid random variables in $\Zbb_7$ with uniform distribution
    and $Y_i=\E\sbra{f(\bm x)\middle|x_1,\cdots,x_i},i=0,1,\cdots,n$.
    Then $Y_i$ is a martingale. Since
    $$
    \abs{Y_i-Y_{i-1}}=
    \abs{\E_{x_i',\cdots,x_n'}\sbra{f(x_1,\cdots,x_i,x_i',\cdots,x_n')-f(x_1,\cdots,x_{i-1},x_i',\cdots,x_n')}}
    \leq1.
    $$
    Thus by Azuma's Inequality, we have
    $$
    \Pr\sbra{Y_n-Y_0<-c\sqrt{n}}<e^{-c^2/2}.
    $$
    Since
    $$
    \Pr\sbra{Y_n<Y_0-2\sqrt{n}}<e^{-2}<\frac17=\Pr\sbra{Y_n=0},
    $$
    then $Y_0\leq2\sqrt n$. Hence by Azuma's Inequality again, we have
    $$
    \Pr\sbra{Y_n>(c+2)\sqrt n}\leq\Pr\sbra{Y_n>Y_0+c\sqrt n}\leq e^{-c^2/2},
    $$
    which means $|W|\leq7^ne^{-c^2/2}$ as desired.
\end{exercise}
